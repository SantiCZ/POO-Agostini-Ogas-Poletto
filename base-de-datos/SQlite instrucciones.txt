1) abrir cmd : cd C:\sqlite
2) sqlite3.exe tasty_alcancia.db creo el archivo .db para mi base de datos
3) activo las claves foraneas: PRAGMA foreign_keys = ON;
4) creo lass tablas: 

CREATE TABLE usuarios (
    id_usuario INTEGER PRIMARY KEY AUTOINCREMENT,
    nombre TEXT NOT NULL,
    email TEXT NOT NULL UNIQUE,
    clave TEXT NOT NULL,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    actividad INTEGER DEFAULT 1
);

CREATE TABLE categorias (
    id_categoria INTEGER PRIMARY KEY AUTOINCREMENT,
    id_usuario INTEGER,
    nombre TEXT NOT NULL,
    icono TEXT,
    color TEXT,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

CREATE TABLE comprobantes (
    id_comprobante INTEGER PRIMARY KEY AUTOINCREMENT,
    id_usuario INTEGER NOT NULL,
    ruta_archivo TEXT NOT NULL,
    fecha_subida DATETIME DEFAULT CURRENT_TIMESTAMP,
    estado TEXT DEFAULT 'pendiente'
        CHECK (estado IN ('pendiente', 'procesado', 'error')),
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

CREATE TABLE gastos (
    id_gasto INTEGER PRIMARY KEY AUTOINCREMENT,
    id_usuario INTEGER NOT NULL,
    id_comprobante INTEGER,
    id_categoria INTEGER NOT NULL,
    comercio TEXT,
    monto REAL NOT NULL,
    fecha_gasto DATE NOT NULL,
    notas TEXT,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (id_comprobante) REFERENCES comprobantes(id_comprobante)
        ON DELETE SET NULL
        ON UPDATE CASCADE,
    FOREIGN KEY (id_categoria) REFERENCES categorias(id_categoria)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

CREATE TABLE items_gasto (
    id_item INTEGER PRIMARY KEY AUTOINCREMENT,
    id_gasto INTEGER NOT NULL,
    descripcion TEXT NOT NULL,
    cantidad REAL DEFAULT 1,
    precio_unitario REAL NOT NULL,
    subtotal REAL NOT NULL,
    FOREIGN KEY (id_gasto) REFERENCES gastos(id_gasto)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

CREATE TABLE suscripciones (
    id_suscripcion INTEGER PRIMARY KEY AUTOINCREMENT,
    id_usuario INTEGER NOT NULL,
    id_categoria INTEGER NOT NULL,
    nombre TEXT NOT NULL,
    monto REAL NOT NULL,
    moneda TEXT DEFAULT 'ARS',
    frecuencia TEXT DEFAULT 'mensual'
        CHECK (frecuencia IN ('diaria', 'semanal', 'mensual', 'anual')),
    vencimiento DATE NOT NULL,
    alerta INTEGER DEFAULT 3,
    actividad INTEGER DEFAULT 1,
    notas TEXT,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON DELETE CASCADE
        ON UPDATE CASCADE,
    FOREIGN KEY (id_categoria) REFERENCES categorias(id_categoria)
        ON DELETE RESTRICT
        ON UPDATE CASCADE
);

CREATE TABLE notificaciones (
    id_notificacion INTEGER PRIMARY KEY AUTOINCREMENT,
    id_usuario INTEGER NOT NULL,
    tipo TEXT NOT NULL
        CHECK (tipo IN ('vencimiento', 'resumen', 'alerta')),
    mensaje TEXT NOT NULL,
    leida INTEGER DEFAULT 0,
    fecha_creacion DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES usuarios(id_usuario)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

5) verifico que se hayan creado: .tables
6) verificar si un insert se hizo bien : select * from usuarios
6) salir: .quit
